clear           → shell clear

login           → shell gh auth status || gh auth login

eval-all-record → for n in 1 to 112:
                    init n --silent;
                    eval all --record runs=10;
                    stack --process-deletions --silent;
                    reset --discard-changes --silent;
                    echo recorded problem n;

eval-all-show   → for n in 1 to 112:
                    init n --silent;
                    eval dev main --show;
                    reset --discard-changes --silent;
                    echo evaluated problem n;

migrate-all     → for n in 1 to 112:
                    init n --silent;
                    migrate;
                    eval;
                    reset --silent;
                    echo migrated problem n;

reinit-all      → for n in 1 to 996:
                    init n --reinit;
                    stack --process-deletions --silent;
                    reset --discard-changes --silent;
                    echo re-initialized problem n;
